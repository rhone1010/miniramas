// app/api/analyze/route.ts
// UPDATE: Accepts multiple images. All analyzed together into one merged blueprint.
// Primary image defines composition. Additional images validate structure only.

import { NextRequest, NextResponse } from 'next/server'
import { extractBlueprintMulti } from '@/lib/structure/extractBlueprint'
import { buildStructureLock, buildAntiDrift } from '@/lib/structure/buildStructureLock'

export const maxDuration = 60

export async function POST(request: NextRequest) {
  try {
    const rawText = await request.text()
    let body: {
      image_b64?:   string    // single image (backwards compat)
      images_b64?:  string[]  // multi-image array [primary, secondary?, tertiary?]
      image_url?:   string
      image_id?:    string
    }
    try {
      body = JSON.parse(rawText)
    } catch {
      return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 })
    }

    const openaiApiKey = process.env.OPENAI_API_KEY
    if (!openaiApiKey) {
      return NextResponse.json({ error: 'OPENAI_API_KEY not set' }, { status: 500 })
    }

    // Build images array — support both single and multi
    let images: string[] = []

    if (body.images_b64?.length) {
      // Multi-image path
      images = body.images_b64.slice(0, 3)
    } else if (body.image_b64) {
      // Single image backwards compat
      images = [body.image_b64]
    } else if (body.image_url) {
      const imgRes = await fetch(body.image_url)
      if (!imgRes.ok) {
        return NextResponse.json(
          { error: `Failed to fetch image: ${imgRes.status}` },
          { status: 400 }
        )
      }
      images = [Buffer.from(await imgRes.arrayBuffer()).toString('base64')]
    } else {
      return NextResponse.json(
        { error: 'Provide image_b64, images_b64, or image_url' },
        { status: 400 }
      )
    }

    // Strip any accidental data URL prefixes
    images = images.map(img => img.includes(',') ? img.split(',')[1] : img)

    if (images.some(img => !img || img.length < 100)) {
      return NextResponse.json({ error: 'One or more images is empty or too short' }, { status: 400 })
    }

    const sourceId = body.image_id || `img_${Date.now()}`

    console.log(`[analyze] Processing ${images.length} image(s) for ${sourceId}`)

    // Extract and merge blueprint from all images
    const blueprint    = await extractBlueprintMulti(images, sourceId, openaiApiKey)
    const structureLock = buildStructureLock(blueprint)
    const antiDrift    = buildAntiDrift()

    const multiConfirmed = blueprint.anchors.filter(a => a.confirmed_by?.length > 1).length
    console.log(
      `[analyze] Done — ${blueprint.anchors.length} anchors, ` +
      `${multiConfirmed} multi-confirmed, ` +
      `${blueprint.semantic_notes.length} semantic notes quarantined`
    )

    const topAnchors = blueprint.anchors
      .filter(a => a.type === 'structure')
      .slice(0, 5)
      .map(a => a.description)

    return NextResponse.json({
      blueprint,
      structure_lock: structureLock,
      anti_drift:     antiDrift,
      summary: {
        anchor_count:     blueprint.anchors.length,
        multi_confirmed:  multiConfirmed,
        image_count:      images.length,
        top_anchors:      topAnchors,
        non_negotiables:  topAnchors,
        semantic_notes:   blueprint.semantic_notes,
      },
    })

  } catch (error) {
    console.error('[analyze] Error:', error)
    return NextResponse.json(
      { error: error instanceof Error ? error.message : 'Analysis failed' },
      { status: 500 }
    )
  }
}
